# High-End Expose Generator — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Auto-generate professional real estate exposes (PDF) from property data, images, and unit information — matching InDesign quality using Puppeteer/Chromium rendering.

**Architecture:** HTML templates rendered to PDF via Puppeteer (headless Chromium). Two template types: Standard (Wohnung/Haus, Portrait A4) and Neubauprojekt (Landscape A4). Missing descriptions auto-generated by Claude AI. Data collected from properties, property_images, property_units, admin_settings tables.

**Tech Stack:** Laravel 11, Blade templates, Puppeteer (via spatie/browsershot or direct Node.js), Anthropic API (for missing descriptions), Google Fonts (Outfit + Playfair Display)

```javascript
#!/usr/bin/env node
const puppeteer = require('puppeteer');
const fs = require('fs');

const [,, htmlPath, outputPath, orientation = 'portrait'] = process.argv;

if (!htmlPath || !outputPath) {
  console.error('Usage: render-pdf.js <html-path> <output-path> [portrait|landscape]');
  process.exit(1);
}

(async () => {
  const browser = await puppeteer.launch({
    args: ['--no-sandbox', '--disable-setuid-sandbox'],
    headless: 'new',
  });
  const page = await browser.newPage();

  const html = fs.readFileSync(htmlPath, 'utf8');
  await page.setContent(html, { waitUntil: 'networkidle0', timeout: 30000 });

  await page.pdf({
    path: outputPath,
    format: 'A4',
    landscape: orientation === 'landscape',
    printBackground: true,
    preferCSSPageSize: true,
    margin: { top: 0, right: 0, bottom: 0, left: 0 },
  });

  await browser.close();
  console.log(JSON.stringify({ success: true, path: outputPath }));
})().catch(err => {
  console.error(JSON.stringify({ success: false, error: err.message }));
  process.exit(1);
});
```

- [ ] **Step 3: Create PdfRenderService.php**

Create `app/Services/PdfRenderService.php`:
```php
<?php
namespace App\Services;

use Illuminate\Support\Facades\Log;

class PdfRenderService
{
    public function renderHtmlToPdf(string $html, string $outputPath, string $orientation = 'portrait'): bool
    {
        // Write HTML to temp file
        $tmpHtml = storage_path('app/tmp_expose_' . uniqid() . '.html');
        file_put_contents($tmpHtml, $html);

        $scriptPath = base_path('scripts/render-pdf.js');
        $cmd = sprintf(
            'node %s %s %s %s 2>&1',
            escapeshellarg($scriptPath),
            escapeshellarg($tmpHtml),
            escapeshellarg($outputPath),
            escapeshellarg($orientation)
        );

        $output = shell_exec($cmd);
        @unlink($tmpHtml);

        $result = json_decode($output, true);
        if (!$result || empty($result['success'])) {
            Log::error('PDF render failed', ['output' => $output]);
            return false;
        }

        return file_exists($outputPath);
    }
}
```

Create `app/Services/ExposeDataService.php`:
```php
<?php
namespace App\Services;

use Illuminate\Support\Facades\DB;
use Illuminate\Support\Facades\Log;

class ExposeDataService
{
    public function collectExposeData(int $propertyId): array
    {
        $property = DB::table('properties')->where('id', $propertyId)->first();
        if (!$property) throw new \Exception("Property {$propertyId} not found");

        $images = DB::table('property_images')
            ->where('property_id', $propertyId)
            ->orderByDesc('is_title_image')
            ->orderBy('sort_order')
            ->get();

        $units = DB::table('property_units')
            ->where('property_id', $propertyId)
            ->where('is_parking', 0)
            ->orderByRaw("floor ASC, CAST(REPLACE(unit_number, 'TOP ', '') AS UNSIGNED) ASC")
            ->get();

        $parking = DB::table('property_units')
            ->where('property_id', $propertyId)
            ->where('is_parking', 1)
            ->where('status', 'frei')
            ->count();

        $broker = $this->getBrokerData($property->broker_id ?? 1);
        $owner = $property->customer_id
            ? DB::table('customers')->where('id', $property->customer_id)->first()
            : null;

        $isNeubauprojekt = $this->isNeubauprojekt($property);
        $nebenkosten = $this->calculateNebenkosten($property);

        // Generate missing descriptions via AI
        $descriptions = $this->ensureDescriptions($property);

        // Group images by category
        $imagesByCategory = $this->groupImages($images);

        // Group units by floor
        $unitsByFloor = $this->groupUnitsByFloor($units);

        // Floor plan images (category = 'grundriss')
        $floorPlans = $images->where('category', 'grundriss');

        // Logo as base64
        $logoPath = public_path('assets/logo-full-orange.png');
        $logoBase64 = file_exists($logoPath)
            ? 'data:image/png;base64,' . base64_encode(file_get_contents($logoPath))
            : null;

        // Broker photo as base64
        $brokerPhotoBase64 = null;
        if (!empty($broker['photo_path'])) {
            $photoFullPath = storage_path('app/public/' . $broker['photo_path']);
            if (file_exists($photoFullPath)) {
                $brokerPhotoBase64 = 'data:image/jpeg;base64,' . base64_encode(file_get_contents($photoFullPath));
            }
        }

        return [
            'property'           => $property,
            'images'             => $images,
            'imagesByCategory'   => $imagesByCategory,
            'titleImage'         => $images->firstWhere('is_title_image', 1),
            'floorPlans'         => $floorPlans,
            'units'              => $units,
            'unitsByFloor'       => $unitsByFloor,
            'parkingCount'       => $parking,
            'broker'             => $broker,
            'brokerPhotoBase64'  => $brokerPhotoBase64,
            'owner'              => $owner,
            'isNeubauprojekt'    => $isNeubauprojekt,
            'nebenkosten'        => $nebenkosten,
            'descriptions'       => $descriptions,
            'logoBase64'         => $logoBase64,
            'generatedAt'        => now()->format('d.m.Y'),
            'orientation'        => $isNeubauprojekt ? 'landscape' : 'portrait',
        ];
    }

    private function isNeubauprojekt($property): bool
    {
        $type = strtolower($property->object_type ?? '');
        return str_contains($type, 'neubau') || str_contains($type, 'projekt');
    }

    private function getBrokerData(?int $brokerId): array
    {
        $userId = $brokerId ?? 1;
        $user = DB::table('users')->where('id', $userId)->first();
        $settings = DB::table('admin_settings')->where('user_id', $userId)->first();

        return [
            'name'       => $settings->signature_name ?? $user->name ?? 'SR-Homes',
            'title'      => $settings->signature_title ?? '',
            'company'    => $settings->signature_company ?? 'SR-Homes Immobilien GmbH',
            'phone'      => $settings->signature_phone ?? '',
            'email'      => $user->email ?? 'office@sr-homes.at',
            'website'    => $settings->signature_website ?? 'www.sr-homes.at',
            'photo_path' => $settings->signature_photo_path ?? null,
        ];
    }

    private function calculateNebenkosten($property): array
    {
        $price = (float) ($property->purchase_price ?? 0);
        if ($price <= 0) return [];

        $grest = $price * 0.035;
        $grundbuch = $price * 0.011;
        $notar = $price * 0.015;
        $provisionPct = (float) ($property->buyer_commission_percent ?? 3.0);
        $provision = $price * ($provisionPct / 100) * 1.20; // inkl. 20% USt

        return [
            'kaufpreis'    => $price,
            'grest'        => $grest,
            'grest_pct'    => '3,5%',
            'grundbuch'    => $grundbuch,
            'grundbuch_pct'=> '1,1%',
            'notar'        => $notar,
            'notar_pct'    => 'ca. 1,5%',
            'provision'    => $provision,
            'provision_pct'=> number_format($provisionPct, 1, ',', '') . '% + 20% USt',
            'gesamt'       => $price + $grest + $grundbuch + $notar + $provision,
        ];
    }

    private function ensureDescriptions($property): array
    {
        $desc = [
            'objekt' => $property->realty_description ?? '',
            'lage'   => $property->location_description ?? '',
            'ausstattung' => $property->equipment_description ?? '',
            'sonstiges' => $property->other_description ?? '',
            'highlights' => $property->highlights ?? '',
        ];

        // If main description is empty, generate via AI
        if (empty(trim($desc['objekt']))) {
            try {
                $anthropic = app(AnthropicService::class);
                $prompt = $this->buildDescriptionPrompt($property);
                $result = $anthropic->chat(
                    'Du bist ein professioneller Immobilien-Texter. Schreibe eine hochwertige, sachliche Objektbeschreibung fuer ein Makler-Expose auf Deutsch. Keine Emojis, kein Marketing-Blabla. Max 150 Woerter.',
                    $prompt,
                    800
                );
                if ($result) $desc['objekt'] = $result;
            } catch (\Throwable $e) {
                Log::warning('Expose: AI description generation failed: ' . $e->getMessage());
            }
        }

        if (empty(trim($desc['lage'])) && !empty($property->city)) {
            try {
                $anthropic = app(AnthropicService::class);
                $result = $anthropic->chat(
                    'Du bist ein professioneller Immobilien-Texter. Schreibe eine sachliche Lagebeschreibung fuer ein Expose. Max 100 Woerter. Deutsch.',
                    "Objekt in {$property->address}, {$property->zip} {$property->city}. Objekttyp: {$property->object_type}.",
                    500
                );
                if ($result) $desc['lage'] = $result;
            } catch (\Throwable $e) {}
        }

        return $desc;
    }

    private function buildDescriptionPrompt($property): string
    {
        $parts = ["Erstelle eine Objektbeschreibung fuer:"];
        if ($property->title) $parts[] = "Titel: {$property->title}";
        if ($property->object_type) $parts[] = "Typ: {$property->object_type}";
        if ($property->address) $parts[] = "Adresse: {$property->address}, {$property->city}";
        if ($property->living_area) $parts[] = "Wohnflaeche: {$property->living_area} m2";
        if ($property->rooms_amount) $parts[] = "Zimmer: {$property->rooms_amount}";
        if ($property->construction_year) $parts[] = "Baujahr: {$property->construction_year}";
        if ($property->realty_condition) $parts[] = "Zustand: {$property->realty_condition}";
        if ($property->heating) $parts[] = "Heizung: {$property->heating}";
        return implode("\n", $parts);
    }

    private function groupImages($images): array
    {
        $groups = [];
        foreach ($images as $img) {
            $cat = $img->category ?? 'sonstiges';
            $groups[$cat][] = $img;
        }
        return $groups;
    }

    private function groupUnitsByFloor($units): array
    {
        $floors = [];
        $floorLabels = [0 => 'Erdgeschoss', 1 => '1. Obergeschoss', 2 => '2. Obergeschoss', 3 => 'Dachgeschoss'];

        foreach ($units as $unit) {
            $floor = $unit->floor ?? 0;
            $label = $floorLabels[$floor] ?? "{$floor}. Geschoss";
            if (!isset($floors[$floor])) {
                $floors[$floor] = ['label' => $label, 'units' => []];
            }
            $floors[$floor]['units'][] = $unit;
        }

        ksort($floors);
        return $floors;
    }
}
```

- [ ] **Step 1: Add controller methods to PropertyController**

Add three methods:
- `generateExpose(Request)` — Collects data, renders Blade, converts to PDF via Puppeteer, saves to `storage/app/public/exposes/{property_id}/`, returns JSON with URL

```php
public function generateExpose(Request $request): JsonResponse
{
    $propertyId = intval($request->query('property_id', 0));
    if (!$propertyId) return response()->json(['error' => 'property_id required'], 400);

    try {
        $dataService = app(ExposeDataService::class);
        $data = $dataService->collectExposeData($propertyId);

        $template = $data['isNeubauprojekt'] ? 'exposes.neubauprojekt' : 'exposes.standard';
        $html = view($template, $data)->render();

        $dir = storage_path('app/public/exposes/' . $propertyId);
        if (!is_dir($dir)) mkdir($dir, 0755, true);
        $filename = 'Expose_' . preg_replace('/[^a-zA-Z0-9_-]/', '_', $data['property']->address ?? 'Objekt') . '_' . date('Y-m-d') . '.pdf';
        $outputPath = $dir . '/' . $filename;

        $pdfService = app(PdfRenderService::class);
        $success = $pdfService->renderHtmlToPdf($html, $outputPath, $data['orientation']);

        if (!$success) return response()->json(['error' => 'PDF generation failed'], 500);

        // Also save as property_file
        $relativePath = 'exposes/' . $propertyId . '/' . $filename;
        DB::table('property_files')->insert([
            'property_id' => $propertyId,
            'label' => 'Expose ' . date('d.m.Y'),
            'filename' => $filename,
            'path' => $relativePath,
            'mime_type' => 'application/pdf',
            'file_size' => filesize($outputPath),
            'sort_order' => 0,
            'created_at' => now(),
        ]);

        return response()->json([
            'success' => true,
            'url' => url('/storage/' . $relativePath),
            'filename' => $filename,
        ]);
    } catch (\Throwable $e) {
        Log::error('Expose generation failed: ' . $e->getMessage());
        return response()->json(['error' => $e->getMessage()], 500);
    }
}
```
